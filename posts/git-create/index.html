Из этого поста ты узнаешь как создать локально git репозиторий и опубликовать его на GiThub

mkdir Yt-Termux #coздание директории
cd Yt-Termux #заходим туда
echo "# Yt-Termux" >> README.md #создаём RE...
git init #инициализируем
git add README.md #добавляем
git commit -m "первый коммит"
git branch -M main
git remote add origin git@github.com:MarchBro/Yt-Termux.git
git push -u origin main
